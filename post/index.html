  • 有一個地方是原題漏掉的，不知道大家看到有沒有覺得奇怪

char *what = GREET; 那這個GREET是什麼東東

後來發現漏掉一行程式碼： #define GREET “Welcome, I am pleased to meet you“

  • //這段程式碼驗證有沒有 ExploitEducation的環境變數，沒的話則exit 
ptr = getenv("ExploitEducation");
if (NULL == ptr) {
  // This style of comparison prevents issues where you may accidentally
  // type if(ptr = NULL) {}..
  errx(1, "Please specify an environment variable called ExploitEducation");
}
    • 沒的話就給一個吧：
    • export ExploitEducation= $(cyclic 140)
    • env | grep ExploitEducation
  • // 這段程式碼將maxSize大小限制在127
char buffer[128];
int maxSize;
maxSize = strlen(who);
if (maxSize > (sizeof(buffer) - /* ensure null termination */ 1)) {
  maxSize = sizeof(buffer) - 1;
}

  • 有問題的程式碼在哪邊？？ 有sense的觀眾應該猜得到跟下面兩行有關XD

strcpy(buffer, what);

strncpy(buffer + strlen(buffer), who, maxSize);

strcpy() 將GREET message複製至buffer，strncpy()再將who的127 bytes複製至已放入GREET message的buffer中
